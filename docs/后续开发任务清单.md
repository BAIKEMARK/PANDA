# backend/app/shared/infrastructure/redis_client.py
import redis.asyncio as redis
from backend.app.config.config import settings

class RedisClient:
    """Redis客户端 - 单例模式"""

    def __init__(self):
        self.redis = redis.from_url(
            settings.REDIS_URL,
            encoding="utf-8",
            decode_responses=True
        )

    async def get_patient_state(self, session_id: str) -> dict:
        """获取病人实时状态"""
        key = f"patient:state:{session_id}"
        state = await self.redis.hgetall(key)
        return state if state else None

    async def update_patient_state(self, session_id: str, updates: dict):
        """更新病人实时状态"""
        key = f"patient:state:{session_id}"
        await self.redis.hset(key, mapping=updates)
        await self.redis.expire(key, settings.REDIS_TTL_STATE)  # 24小时TTL

    async def save_conversation_snapshot(self, session_id: str, snapshot: dict):
        """保存对话历史快照（最近50轮）"""
        key = f"patient:history:{session_id}"
        await self.redis.lpush(key, json.dumps(snapshot))
        await self.redis.ltrim(key, 0, settings.REDIS_MAX_HISTORY - 1)  # 保留50轮
```

# backend/app/modules/chat/services/patient_state_service.py
class PatientStateService:
    """病人状态管理服务 - Redis + MySQL混合存储"""

    def __init__(self, db: Session):
        self.db = db
        self.redis = redis_client

    async def initialize_state(
        self,
        session_id: str,
        scenario_config: dict
    ) -> dict:
        """初始化病人状态（从scenario和skill_config获取初始值）"""
        # 从skill_config.json读取初始值范围
        initial_state = {
            "mood_score": random.randint(40, 60),
            "satisfaction_score": random.randint(30, 50),
            "depression_level": random.randint(50, 70),
            "rapport_score": random.randint(30, 50),
            "message_count": 0,
            "is_crisis": False,
            "last_updated": int(time.time())
        }

        # 写入Redis
        await self.redis.update_patient_state(session_id, initial_state)

        # 同时保存到MySQL（初始快照）
        await self._persist_state(session_id, initial_state)

        return initial_state

    async def get_state(self, session_id: str) -> dict:
        """获取病人状态（优先从Redis）"""
        state = await self.redis.get_patient_state(session_id)

        if not state:
            # Redis不存在，尝试从MySQL恢复（场景复现）
            state = await self._restore_from_mysql(session_id)

        return state

    async def update_state(
        self,
        session_id: str,
        updates: dict
    ) -> dict:
        """更新病人状态"""
        # 1. 获取当前状态
        current = await self.get_state(session_id)

        # 2. 应用更新
        for key, value in updates.items():
            if key.endswith('_delta'):
                # 处理增量变化
                field = key.replace('_delta', '')
                current[field] = max(0, min(100, current.get(field, 50) + value))
            else:
                current[key] = value

        current['last_updated'] = int(time.time())
        current['message_count'] = current.get('message_count', 0) + 1

        # 3. 更新Redis
        await self.redis.update_patient_state(session_id, current)

        # 4. 异步持久化到MySQL（不阻塞响应）
        asyncio.create_task(self._persist_state(session_id, current))

        return current

    async def _persist_state(self, session_id: str, state: dict):
        """持久化到MySQL"""
        state_record = PatientState(
            id=str(uuid.uuid4()),
            session_id=session_id,
            message_count=state.get('message_count', 0),
            mood_score=state.get('mood_score', 50),
            satisfaction_score=state.get('satisfaction_score', 40),
            depression_level=state.get('depression_level', 60),
            rapport_score=state.get('rapport_score', 40),
            is_crisis=state.get('is_crisis', False),
            crisis_type=state.get('crisis_type')
        )
        self.db.add(state_record)
        await self.db.commit()

    async def _restore_from_mysql(self, session_id: str) -> dict:
        """从MySQL恢复状态（场景复现）"""
        # 获取最后一次状态记录
        last_state = await self.db.execute(
            select(PatientState)
            .where(PatientState.session_id == session_id)
            .order_by(PatientState.created_at.desc())
            .limit(1)
        )
        last_state = last_state.scalar_one_or_none()

        if last_state:
            state = {
                "mood_score": last_state.mood_score,
                "satisfaction_score": last_state.satisfaction_score,
                "depression_level": last_state.depression_level,
                "rapport_score": last_state.rapport_score,
                "message_count": last_state.message_count,
                "is_crisis": last_state.is_crisis,
                "crisis_type": last_state.crisis_type
            }
            # 恢复到Redis
            await self.redis.update_patient_state(session_id, state)
            return state

        # 不存在则返回默认状态
        return {}
```

async def resume_session(session_id: str):

    # 1. 从Redis获取状态（可能已过期）
    state = await redis_client.get_patient_state(session_id)

    def __init__(self):
        super().__init__()
        self.prompt_template = create_patient_agent_prompt()
        self.state_calculator = StateCalculator()

    def invoke(
        self,
        scenario_config: dict,
        current_state: PatientState,
        user_message: str,
        conversation_history: list
    ) -> dict:
        """调用Patient-Agent，返回回复+状态更新"""

        # 1. 构建包含当前状态的Prompt
        state_prompt = self._build_state_prompt(current_state)
        system_prompt = f"{scenario_config['system_prompt']}\n{state_prompt}"

        # 2. 调用父类对话链
        response = super().invoke(
            system_prompt=system_prompt,
            user_message=user_message,
            conversation_history=conversation_history
        )

        # 3. 计算状态变化
        state_updates = self.state_calculator.calculate_updates(
            user_message=user_message,
            current_state=current_state
        )

        return {
            "response": response,
            "state_updates": state_updates
        }

### 任务1.5: 对话服务集成Agent (7h)

**文件**: `backend/app/modules/chat/api/chat.py`

**核心流程**（修改现有conversation_engine.py）:
```python
class ConversationEngine:
    """对话编排引擎 - 集成Agent"""

    def __init__(self, ...):
        # 现有代码...
        self.patient_agent = PatientAgentChain()
        self.state_service = PatientStateService(db)
        self.crisis_detector = CrisisDetector()

    def generate_ai_response(
        self,
        session_id: str,
        user_message: str,
        user_id: str = "user-001"
    ) -> dict:
        """生成AI回复（集成Patient-Agent）"""

        # 1. 获取会话和场景配置
        session = self.repository.get_chat_session(session_id)
        scenario_config = self.scenario_interface.get_scenario_config(...)

        # 2. 获取/创建病人状态
        current_state = self.state_service.get_or_create_state(
            session_id=session_id,
            scenario_config=scenario_config
        )

        # 3. 获取对话历史
        conversation_history = self.repository.get_session_messages(session_id)

        # 4. 调用Patient-Agent
        agent_response = self.patient_agent.invoke(
            scenario_config=scenario_config,
            current_state=current_state,
            user_message=user_message,
            conversation_history=conversation_history
        )

        # 5. 更新病人状态
        updated_state = self.state_service.update_state(
            session_id=session_id,
            updates=agent_response["state_updates"]
        )

        # 6. 危机检测
        crisis = self.crisis_detector.check_crisis(updated_state)
        if crisis:
            return self._handle_crisis(session_id, crisis, updated_state)

        # 7. 保存消息
        self.repository.create_message({
            "session_id": session_id,
            "role": "assistant",
            "content": agent_response["response"],
            "meta_data": {
                "state_updates": agent_response["state_updates"],
                "patient_state": updated_state.dict()
            }
        })

        return {"response": agent_response["response"]}
```
