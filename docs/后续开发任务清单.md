#### 1.1 创建病人动态指标表 (patient_states)

**设计思路**: 独立表存储每次会话的病人状态，避免频繁查询chat_sessions的meta_data

**表结构设计**:
```sql
CREATE TABLE `patient_states` (
  `id` CHAR(36) PRIMARY KEY COMMENT '状态记录ID',
  `session_id` CHAR(36) NOT NULL COMMENT '会话ID',
  `message_count` INT DEFAULT 0 COMMENT '对话轮次',

  -- 核心指标 (0-100分制)
  `mood_score` INT DEFAULT 50 COMMENT '心情指数 (0=极度低落, 100=心情愉悦)',
  `satisfaction_score` INT DEFAULT 40 COMMENT '满意度 (0=极度不满, 100=非常满意)',

  -- 状态标记
  `is_crisis` BOOLEAN DEFAULT FALSE COMMENT '是否触发危机状态',
  `crisis_type` VARCHAR(50) DEFAULT NULL COMMENT '危机类型: suicide/self_harm/abandon',
  `last_triggered_action` VARCHAR(100) DEFAULT NULL COMMENT '最后一次触发的动作',

  -- 时间戳
```

```python
# 使用Redis缓存当前会话状态，减少数据库查询
class PatientStateCache:
    """病人状态缓存管理"""

    def __init__(self, redis_client):
        self.redis = redis_client
        self.cache_ttl = 3600  # 1小时过期

    async def get_state(self, session_id: str) -> PatientState:
        """优先从Redis获取，不存在则查询数据库"""
        cache_key = f"patient_state:{session_id}"

        # 1. 尝试从Redis获取
        cached = await self.redis.get(cache_key)
        if cached:
            return PatientState.parse_raw(cached)

        # 2. 从数据库查询
        state = await db.query(PatientState).filter(
            PatientState.session_id == session_id
        ).first()

        # 3. 写入Redis缓存
        await self.redis.setex(
            cache_key,
            self.cache_ttl,
            state.json()
        )

        return state

    async def update_state(self, session_id: str, updates: dict):
        """更新状态（同时更新Redis和数据库）"""
        # 更新数据库
        await db.execute(
            update(PatientState)
            .where(PatientState.session_id == session_id)
            .values(**updates)
        )

        # 更新Redis
        cache_key = f"patient_state:{session_id}"
        state = await self.get_state(session_id)
        await self.redis.setex(
            cache_key,
            self.cache_ttl,
            state.json()
        )
```

#### 2.3 危机检测与强制结束机制

**危机触发条件**:
```python
# backend/app/services/patient_state_service.py

class CrisisDetector:
    """危机检测器"""

    # 危机阈值配置
    CRISIS_THRESHOLDS = {
        "mood_too_low": 15,        # 心情低于15分
        "satisfaction_too_low": 10,  # 满意度低于10分
        "depression_too_high": 85,   # 抑郁程度高于85分
        "rapport_broken": 10         # 信任度低于10分
    }

    @staticmethod
    def check_crisis(state: PatientState) -> Optional[dict]:
        """检查是否触发危机"""

        triggers = []

        # 1. 心意极度低落
        if state.mood_score < CrisisDetector.CRISIS_THRESHOLDS["mood_too_low"]:
            triggers.append({
                "type": "extreme_low_mood",
                "message": "患者心情极度低落，拒绝继续交流",
                "action": "force_end"
            })

        # 2. 满意度崩塌
        if state.satisfaction_score < CrisisDetector.CRISIS_THRESHOLDS["satisfaction_too_low"]:
            triggers.append({
                "type": "dissatisfaction",
                "message": "患者对当前护理方式非常不满",
                "action": "force_end"
            })

        # 3. 信任破裂
        if state.rapport_score < CrisisDetector.CRISIS_THRESHOLDS["rapport_broken"]:
            triggers.append({
                "type": "rapport_broken",
                "message": "患者已失去信任，抗拒交流",
                "action": "force_end"
            })

        # 4. 重度抑郁 (不强制结束，但标记危机)
        if state.depression_level > CrisisDetector.CRISIS_THRESHOLDS["depression_too_high"]:
            triggers.append({
                "type": "severe_depression",
                "message": "患者抑郁程度严重，需立即干预",
                "action": "mark_crisis"
            })

        return triggers if triggers else None

    @staticmethod
    def get_crisis_response(trigger_type: str) -> str:
        """根据危机类型生成响应"""

        responses = {
            "extreme_low_mood": "（沉默片刻...）我想一个人静静...（转头不再说话）",
            "dissatisfaction": "你根本不理解我...算了，不想说了。",
            "rapport_broken": "...（低头不语，表现出明显的抗拒）"
        }

        return responses.get(trigger_type, "（患者情绪异常，对话无法继续）")
```

```python
    message_data: ChatMessageCreate,

                "crisis_triggered": True,
                "crisis_type": crisis[0]["type"],
                "final_state": updated_state.dict()
            }
        )

    # 6. 正常返回AI回复
    return service.create_message(...)
```

**关键词检测逻辑**:
```python
# backend/app/services/scale_trigger_service.py

class ScaleTriggerService:
    """量表触发服务"""

    def __init__(self, db: Session):
        self.db = db

    async def detect_scale_trigger(
        self,
        user_message: str,
        session_state: PatientState
    ) -> Optional[dict]:
        """检测是否应该触发量表"""

        # 1. 关键词匹配
        detected_scale = self._match_keywords(user_message)
        if not detected_scale:
            return None

        # 2. 检查是否已经发送过该量表
        if await self._already_sent(session_state.session_id, detected_scale):
            return {
                "action": "remind",
                "message": f"已经发送过{detected_scale}量表了，请查看。",
                "scale": detected_scale
            }

        # 3. 检查对话轮次（建议至少3轮后再触发）
        if session_state.message_count < 3:
            return {
                "action": "defer",
                "message": "我们先多聊一聊，稍后再做评估。",
                "scale": detected_scale
            }

        # 4. 检查患者状态（如果心情太低，不适合做量表）
        if session_state.mood_score < 25:
            return {
                "action": "comfort_first",
                "message": "我看你现在情绪很低落，我们先聊聊感受，不用急着做测试。",
                "scale": detected_scale
            }

        # 5. 通过所有检查，触发量表
        return {
            "action": "trigger",
            "message": f"好的，我准备了一份{detected_scale}量表，可以帮助我们更好地了解你的情况。",
            "scale": detected_scale
        }

    def _match_keywords(self, message: str) -> Optional[str]:
        """关键词匹配"""
        message_lower = message.lower()

        for scale_type, keywords in SCALE_TRIGGER_KEYWORDS.items():
            for keyword in keywords:
                if keyword in message_lower:
                    return scale_type

        return None

    async def _already_sent(self, session_id: str, scale_type: str) -> bool:
        """检查是否已发送过"""
        count = await self.db.query(ScaleTrigger).filter(
            ScaleTrigger.session_id == session_id,
            ScaleTrigger.scale_type == scale_type
        ).count()

        return count > 0
```

```python
# backend/app/services/dify_client.py

import httpx
from typing import Optional, Dict, Any
from backend.app.core.config import settings

class DifyClient:
    """Dify平台客户端"""

    def __init__(self):
        self.base_url = settings.DIFY_API_URL
        self.api_key = settings.DIFY_API_KEY
        self.timeout = 30

    async def call_patient_agent(
        self,
        user_message: str,
        scenario: dict,
        current_state: PatientState,
        conversation_history: list
    ) -> dict:
        """调用Patient-Agent工作流"""

        url = f"{self.base_url}/v1/workflows/run"

        payload = {
            "inputs": {
                "user_message": user_message,
                "scenario": scenario,
                "current_state": current_state.dict(),
                "conversation_history": [
                    {"role": msg.role, "content": msg.content}
                    for msg in conversation_history
                ]
            },
            "response_mode": "blocking",  # 或 "streaming" 支持流式
            "user": "panda-user"
        }

        async with httpx.AsyncClient() as client:
            response = await client.post(
                url,
                json=payload,
                headers={
                    "Authorization": f"Bearer {self.api_key}",
                    "Content-Type": "application/json"
                },
                timeout=self.timeout
            )
            response.raise_for_status()

            result = response.json()

            # 解析Dify返回的数据
            return {
                "assistant_response": result["data"]["outputs"]["answer"],
                "state_updates": result["data"]["outputs"]["state_updates"],
                "triggered_actions": result["data"]["outputs"].get("triggered_actions", []),
                "metadata": result["data"]["outputs"].get("metadata", {})
            }

    async def call_mentor_agent(
        self,
        conversation_history: list,
        patient_state: PatientState,
        scenario: dict,
        rubric: str
    ) -> dict:
        """调用Mentor-Agent工作流"""

        url = f"{self.base_url}/v1/workflows/run"

        payload = {
            "inputs": {
                "conversation_history": conversation_history,
                "patient_final_state": patient_state.dict(),
                "scenario": scenario,
                "scoring_rubric": rubric
            },
            "response_mode": "blocking",
            "user": "panda-user"
        }

        async with httpx.AsyncClient() as client:
            response = await client.post(
                url,
                json=payload,
                headers={
                    "Authorization": f"Bearer {self.api_key}",
                    "Content-Type": "application/json"
                },
                timeout=60  # 评估可能需要更长时间
            )
            response.raise_for_status()

            result = response.json()

            return result["data"]["outputs"]

    async def stream_patient_agent(
        self,
        user_message: str,
        scenario: dict,
        current_state: PatientState,
        conversation_history: list
    ):
        """流式调用Patient-Agent（用于SSE）"""

        url = f"{self.base_url}/v1/workflows/run"

        payload = {
            "inputs": {
                "user_message": user_message,
                "scenario": scenario,
                "current_state": current_state.dict(),
                "conversation_history": conversation_history
            },
            "response_mode": "streaming",
            "user": "panda-user"
        }

        async with httpx.AsyncClient() as client:
            async with client.stream(
                "POST",
                url,
                json=payload,
                headers={
                    "Authorization": f"Bearer {self.api_key}",
                    "Content-Type": "application/json"
                },
                timeout=self.timeout
            ) as response:
                async for line in response.aiter_lines():
                    if line.startswith("data:"):
                        data = json.loads(line[5:].strip())
                        yield data
```

```python
async def create_chat_message(

    # 3. 获取当前病人状态

    )

    # 4. 获取场景配置
    scenario = scenario_service.get_scenario(session.scenario_id)

    # 5. 获取对话历史
    conversation_history = chat_service.get_session_messages(
        message_data.session_id
    )

    # 6. 调用Dify Patient-Agent
    try:
        dify_response = await dify_client.call_patient_agent(

                "system_prompt": scenario.system_prompt,
                "patient_background": scenario.patient_background,

            },
            current_state=current_state,
            conversation_history=conversation_history
        )
    except Exception as e:
        raise HTTPException(
            status_code=500,
            detail=f"AI服务调用失败: {str(e)}"
        )

    # 7. 更新病人状态
    updated_state = await patient_state_service.update_state(
        session_id=message_data.session_id,
        updates=dify_response["state_updates"]
    )

    # 8. 危机检测
    crisis = crisis_detector.check_crisis(updated_state)
    if crisis:
        # 强制结束会话
        await chat_service.end_session(
            session_id=message_data.session_id,
            final_score=0,
            reason="crisis_triggered"
        )

        # 返回危机响应
        return ChatMessageResponse(
            role="assistant",
            content=crisis_detector.get_crisis_response(crisis[0]["type"]),
            meta_data={
                "crisis_triggered": True,
                "crisis_type": crisis[0]["type"],
                "final_state": updated_state.dict()
            }
        )

    # 9. 检查是否触发了量表
    triggered_action = dify_response.get("triggered_actions", [])
    scale_trigger = next(
        (a for a in triggered_action if a.get("type") == "scale"),
        None
    )

    if scale_trigger:
        # 记录量表触发
        await scale_trigger_service.record_trigger(
            session_id=message_data.session_id,
            scale_type=scale_trigger["scale_type"],
            user_message=message_data.content
        )

    # 10. 保存AI回复
    assistant_message = chat_service.create_message(
        ChatMessageCreate(
            session_id=message_data.session_id,
            role="assistant",
            content=dify_response["assistant_response"],
            meta_data={
                "state_updates": dify_response["state_updates"],
                "triggered_actions": triggered_action,
                "patient_state": updated_state.dict()
            }
        )
    )

    return assistant_message
```
